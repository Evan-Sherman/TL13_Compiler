program
   var SMALLER as int ;
   var BIGGER as int ;
   var TEMP as int ;
   var BOOLEAN as bool ;
   var HELLO as int ;
   var TEMP as bool ;
begin

   BIGGER := readInt ;
   SMALLER := readInt ;
   TEMP := 32 ;
   if SMALLER > BIGGER then
      TEMP := SMALLER ;
      SMALLER := BIGGER ;
      BIGGER := TEMP ;
   end ;

   while SMALLER > 0 do
      BIGGER := BIGGER - SMALLER ;

      if SMALLER > BIGGER then
         TEMP := SMALLER ;
         SMALLER := BIGGER ;
         BIGGER := TEMP ;
      end ;
   end ;
   writeInt BIGGER ;
end
